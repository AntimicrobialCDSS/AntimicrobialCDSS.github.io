function createODTable(selectedItem) { // Check if selectedItem is already an object or a string const itemData = typeof selectedItem === 'string' ? odData.find(item => item.Name === selectedItem) : selectedItem; if (!itemData) { console.error('No matching data found for selected item:', selectedItem); return; } const table = document.createElement('table'); table.classList.add('table'); const tbody = document.createElement('tbody'); // Create the header row with a bold header and a background color const headerRow = document.createElement('tr'); const headerCell = document.createElement('td'); headerCell.textContent = 'ORDER DIALOG INFORMATION'; headerCell.colSpan = 1; // Span across one column headerCell.style.fontFamily = 'PT Serif Bold, Arial, sans-serif'; headerCell.style.fontWeight = 'bold'; headerCell.style.backgroundColor = 'lightblue'; // Set background color headerRow.appendChild(headerCell); tbody.appendChild(headerRow); // Iterate over the entries of itemData Object.entries(itemData).forEach(([key, value]) => { const row = document.createElement('tr'); // Create cells for key and value const cell = document.createElement('td'); if (key.startsWith('Component')) { // Check if there's a matching item in odData for the value const matchingItem = odData.find(item => item.Name.trim().toLowerCase() === value.trim().toLowerCase()); if (matchingItem) { // Make the value clickable const linkElement = document.createElement('a'); linkElement.classList.add('clickable'); linkElement.textContent = value; linkElement.href = '#'; // Prevent default link behavior linkElement.addEventListener('click', () => { createODTable(matchingItem); // Call createODTable for the matching item }); cell.innerHTML = `${key}: `; cell.appendChild(linkElement); } else { cell.innerHTML = `${key}: ${value}`; // No match found, just display the value } } else { cell.innerHTML = `${key}: ${value}`; // For other fields } row.appendChild(cell); tbody.appendChild(row); }); // Append tbody to the table table.appendChild(tbody); // Assuming there's a DOM element with id 'table-container' to hold the table document.getElementById('table-container').appendChild(table); }
